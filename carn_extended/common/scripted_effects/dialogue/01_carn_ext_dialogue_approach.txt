# CARNALITAS EXTENDED — Diyalog Motoru: Katman 1 — Yaklaşma
# File: common/scripted_effects/dialogue/01_carn_ext_dialogue_approach.txt
#
# Her yaklaşma tutumuna göre metin seçer.
# Kullanım: carn_ext_dialogue_approach = { ATTITUDE = dominant }
#
# ATTITUDE seçenekleri:
#   dominant    — karakter kontrol altında, emir verir
#   submissive  — karakter boyun eğiyor, izin istiyor
#   romantic    — duygusal, nazik, şiirsel
#   desperate   — libido baskısı yüksek, ihtiyaç hissediyor
#   forbidden   — yasak ilişki, suçluluk + istek karışımı
#   cold        — duygusuz, mekanik, siyasi
# ==============================================================================

carn_ext_dialogue_approach_dominant = {
    # Dominant karakter yaklaşıyor
    # Root = yaklaşan, scope:partner = hedef
    random_list = {
        1 = {
            # Emir tonu
            custom_tooltip = {
                text = carn_ext_approach_dominant_01
                # "[Root.GetFirstName] yaklaştı, sesi alçak ve kesin:
                #  'Bu gece benim her isteğimi yerine getireceksin.'"
            }
        }
        1 = {
            custom_tooltip = {
                text = carn_ext_approach_dominant_02
                # "Kapıyı kapattı. Sesi neredeyse fısıltıydı — ama emir buyuruyordu:
                #  'Diz çök. Şimdi.'"
            }
        }
        1 = {
            custom_tooltip = {
                text = carn_ext_approach_dominant_03
                # "'Sen benim ne olduğumu biliyorsun,' dedi [Root.GetFirstName].
                #  'Ve bu gece bunun ne anlama geldiğini hatırlatacağım.'"
            }
        }
        1 = {
            # Lustful trait varsa ek seçenek
            trigger = { has_trait = lustful }
            custom_tooltip = {
                text = carn_ext_approach_dominant_lustful_01
                # "Gözleri [partner.GetFirstName]'i baştan aşağı süzdü.
                #  'Ne kadar bekleyebileceğimi sanıyorsun?'"
            }
        }
        1 = {
            # Legendary boyut varsa ek seçenek
            trigger = { has_trait = carn_ext_size_legendary }
            custom_tooltip = {
                text = carn_ext_approach_dominant_legendary_01
                # "'Daha önce böyle bir geceyi hiç yaşamadın,' dedi sakin bir özgüvenle.
                #  'Ama unutmayacaksın.'"
            }
        }
    }
}

            }
        }
        1 = {
            custom_tooltip = {
                text = carn_ext_approach_submissive_03

                #  'Söz veriyorum. İyi bir [Root.GetIsMaleDesc] olacağım bu gece.'"

}

carn_ext_dialogue_approach_romantic = {
    random_list = {
        1 = {
            custom_tooltip = {
                text = carn_ext_approach_romantic_01
                # "'Sizi bu gece çok özledim,' dedi [Root.GetFirstName] sessizce.
                #  'Kelimelerle anlatamıyorum ama belki gösterebilirim.'"
            }
        }
        1 = {
            custom_tooltip = {
                text = carn_ext_approach_romantic_02
                # "Yüzüne uzandı, parmakları yavaşça yanağına değdi:
                #  'Geceyi bize ait kılalım. Başka kimse yok — sadece biz.'"
            }
        }
        1 = {
            custom_tooltip = {
                text = carn_ext_approach_romantic_03
                # "'Her gün yanınızdayım,' dedi.
                #  'Ama bu gece... bu gece gerçekten yanınızda olmak istiyorum.'"
            }
        }
        1 = {
            # Yüksek opinion — daha güçlü aşk tonu
            trigger = {
                opinion = { target = scope:partner  value >= 60 }
            }
            custom_tooltip = {
                text = carn_ext_approach_romantic_deep_01
                # "'Bilmiyorum bunu nasıl söyleyeceğimi,' dedi, sesi kısık:
                #  'Sizi seviyorum. Ve bu gece bunu hissettirmek istiyorum.'"
            }
        }
        1 = {
            # Afterglow modifier varsa — tekrarlanan sevişme
            trigger = {
                has_character_modifier = carn_ext_afterglow
            }
            custom_tooltip = {
                text = carn_ext_approach_romantic_again_01
                # "Sabah henüz gelmemişti. [Root.GetFirstName] dönüp baktı —
                #  'Bir kez daha,' diye fısıldadı. 'Lütfen.'"
            }
        }
    }
}

carn_ext_dialogue_approach_forbidden = {
    # Yasak ilişki — ensest, evli, köle, din adamı vb.
    random_list = {
        1 = {
            custom_tooltip = {
                text = carn_ext_approach_forbidden_01
                # "Uzun süre konuşmadı. Sadece baktı.
                #  Sonunda sesi kısık çıktı: 'Bunu yapmamalıyız. Her ikimiz de biliyoruz.
                #  Ama durduramıyorum kendimi.'"
            }
        }
        1 = {
            custom_tooltip = {
                text = carn_ext_approach_forbidden_02
                # "'Eğer birisi görürse...' diye başladı, durdu.
                #  Sonra başını salladı sanki pes etmiş gibi:
                #  'Umursamıyorum. Şu an tek umursadığım sensin.'"
            }
        }
        1 = {
            custom_tooltip = {
                text = carn_ext_approach_forbidden_03
                # "Fısıltısı neredeyse kayboldu odada:
                #  'Tanrı beni affetsin. Sen [partner_relation]sın.
                #   Ve ben yine de buradayım.'"
            }
        }
        1 = {
            # Güçlü suçluluk — celibate veya dini trait
            trigger = {
                OR = {
                    has_trait = celibate
                    has_trait = devoted
                    piety >= 500
                }
            }
            custom_tooltip = {
                text = carn_ext_approach_forbidden_pious_01
                # "Yıllar süren dualar, yeminler, kefaret...
                #  Hepsi bu an eridi.
                #  'Tanrım,' dedi yalnızca. Ve adım attı.'"
            }
        }
    }
}
